Lesson 9: C Pointers — Detaylı Referans

1. Pointer Temelleri (Tekrar)

type* ptr = &variable; // pointer tanımlama
*ptr // dereference — pointer'ın gösterdiği değere eriş
int x = 42; int* p = &x; // p = x'in adresi printf("%d\n", *p); // 42 *p = 99; // x = 99

2. void* — Generic Pointer

void* herhangi bir türün adresini tutabilir. Tüm dinamik bellek fonksiyonları void* döndürür. Dereference edilemez — önce cast gerekir.

int a = 42; float b = 3.14; void* ptr; ptr = &a; // int* → void* (otomatik) printf("%d\n", *(int*)ptr); // CAST gerekli! ptr = &b; // float* → void* printf("%.2f\n", *(float*)ptr); // 3.14 // void* üzerinde aritmetik YASAK (boyut bilinmiyor) // ptr++; // DERLEME HATASI

3. const ve Pointer'lar — 4 Kombinasyon

İmzaAnlamı
const int* pPointer değişir, değer değişmez (pointer to const int)
int* const pPointer değişmez, değer değişir (const pointer to int)
const int* const pİkisi de değişmez
int* pİkisi de değişir
int x = 10, y = 20; const int* p1 = &x; // değer sabit p1 = &y; // ✅ pointer değişebilir // *p1 = 30; // ❌ DERLEME HATASI int* const p2 = &x; // pointer sabit *p2 = 30; // ✅ değer değişebilir // p2 = &y; // ❌ DERLEME HATASI const int* const p3 = &x; // ikisi de sabit
💡 const'u sağdan sola oku: const int* = "pointer to const int". int* const = "const pointer to int".

4. restrict Keyword (C99)

restrict — pointer'ın bellek bölgesine sadece bu pointer'dan erişileceğini belirtir. Derleyicinin daha iyi optimizasyon yapmasını sağlar.

void copy(int* restrict dest, int* restrict src, int n) { for (int i = 0; i < n; i++) dest[i] = src[i]; // restrict sayesinde derleyici dest ve src'nin asla örtüşmeyeceğini bilir } // DİKKAT: restrict yanlış kullanılırsa tanımsız davranış!

5. Fonksiyon Pointer'ları — Derinlemesine

return_type (*ptr_name)(param_types); // fonksiyon pointer tanımı
#include // 3 farklı fonksiyon int add(int a, int b) { return a + b; } int sub(int a, int b) { return a - b; } int mul(int a, int b) { return a * b; } int main() { // Fonksiyon pointer dizisi (hesap makinesi) int (*ops[])(int, int) = {add, sub, mul}; char op_chars[] = {'+', '-', '*'}; int x = 10, y = 5; for (int i = 0; i < 3; i++) { printf("%d %c %d = %d\n", x, op_chars[i], y, ops[i](x, y)); } return 0; }

5.1 typedef ile Fonksiyon Pointer

// typedef ile daha okunabilir typedef int (*MathOp)(int, int); MathOp op = add; printf("%d\n", op(5, 3)); // 8 // Dizi olarak MathOp ops[] = {add, sub, mul};

5.2 Fonksiyon Pointer ile Callback

// qsort ile sıralama (callback) #include int cmp_int(const void* a, const void* b) { return *(int*)a - *(int*)b; } int main() { int arr[] = {4, 2, 7, 1, 9}; int n = sizeof(arr) / sizeof(arr[0]); qsort(arr, n, sizeof(int), cmp_int); // arr artık {1, 2, 4, 7, 9} return 0; }

6. Pointer Aritmetiği — İleri Seviye

int arr[] = {10, 20, 30, 40, 50}; int* p = arr; printf("%d\n", *(p++)); // 10 (önce *p, sonra p++) — post-increment printf("%d\n", *(++p)); // 30 (önce ++p, sonra *p) — pre-increment printf("%ld\n", p - arr); // 2 (indeks farkı) // p + n: p + n * sizeof(type) byte ilerler // char* ile: +1 = 1 byte // int* ile: +1 = 4 byte

7. Pointer-to-Pointer (int**)

int x = 42; int* p = &x; // pointer to int int** pp = &p; // pointer to pointer to int printf("%d\n", **pp); // 42 // 2D dizi olarak (satır sayısı dinamik) int** matrix = malloc(3 * sizeof(int*)); for (int i = 0; i < 3; i++) matrix[i] = malloc(4 * sizeof(int)); matrix[1][2] = 42; // kullan // Temizlik for (int i = 0; i < 3; i++) free(matrix[i]); free(matrix);

8. Null Pointer ve Hatalar

int* p = NULL; // NULL kontrolü ŞART if (p != NULL) { *p = 5; } else { printf("NULL pointer!\n"); } // NULL = 0 (C'de) if (!p) printf("NULL\n"); // aynı kontrol // Sık hatalar: // ❌ *p = 5; p NULL ise → crash (segmentation fault) // ❌ free(p); free(p); → double free // ❌ int* p; *p = 5; → uninitialized pointer

9. Fonksiyon Pointer Tablosu — Pratik Kullanım

#include // Menü ile farklı işlemler typedef void (*MenuFunc)(); void new_game() { printf("Yeni oyun başladı\n"); } void load_game() { printf("Oyun yükleniyor\n"); } void settings() { printf("Ayarlar\n"); } void quit() { printf("Çıkış\n"); } int main() { MenuFunc menu[] = {new_game, load_game, settings, quit}; int choice; printf("Seçim (0-3): "); scanf("%d", &choice); if (choice >= 0 && choice <= 3) menu[choice](); return 0; }
