Lesson 13: Extra C Topics

1. Macros (#define)

A macro is a name that the preprocessor replaces with a value or code before compilation. No semicolon needed.

#define PI 3.14159 #define MAX 100 #define AREA(r) (PI * (r) * (r)) // Macro with parameter int main() { printf("PI = %f\n", PI); // Becomes: 3.14159 printf("Max = %d\n", MAX); // Becomes: 100 printf("Area = %f\n", AREA(5)); // Becomes: (3.14159 * (5) * (5)) return 0; }
PI = 3.141590
Max = 100
Area = 78.539750
⚠️ Parentheses in macros! Always wrap parameters in (). #define SQUARE(x) x*x with SQUARE(1+2) becomes 1+2*1+2 = 5 (wrong!). Use #define SQUARE(x) ((x)*(x)) instead.

Conditional Compilation

#define DEBUG // Comment this line to disable debug #ifdef DEBUG printf("Debug: x = %d\n", x); #endif #if MAX > 50 printf("MAX is large\n"); #endif

They tell the compiler where a variable lives and how long it lives.

💡 auto is the default (you never write it). register is a suggestion to the compiler. static makes a local variable keep its value between calls.

3. Date & Time ()

#include #include int main() { // Get current time time_t now; time(&now); printf("Current time: %s", ctime(&now)); // Convert to string // Measure how long something takes clock_t start = clock(); for (int i = 0; i < 100000000; i++); // Busy wait clock_t end = clock(); double seconds = (double)(end - start) / CLOCKS_PER_SEC; printf("Loop took: %.3f seconds\n", seconds); return 0; }
Current time: Sat Jul 18 15:30:00 2026
Loop took: 0.245 seconds

4. Random Numbers ()

#include #include #include int main() { // Seed the random generator (once) srand(time(NULL)); // Random numbers printf("Random: %d\n", rand()); // 0 to RAND_MAX (32767) printf("Dice: %d\n", (rand() % 6) + 1); // 1 to 6 printf("0-99: %d\n", rand() % 100); // 0 to 99 return 0; }
💡 Always call srand() once before using rand(). The seed makes the numbers different each run.

5. Error Handling & Debugging

Checking for Errors

#include #include #include int main() { FILE* f = fopen("nonexistent.txt", "r"); if (f == NULL) { printf("Error: %s\n", strerror(errno)); // "No such file or directory" perror("fopen"); // Same thing, shorter return 1; } fclose(f); return 0; }

The NULL Pointer

NULL is a pointer that points to "nothing". Always check for NULL!

int* ptr = NULL; // Not pointing to anything yet if (ptr == NULL) { printf("Pointer is NULL - can't use it!\n"); } // *ptr = 5; // CRASH! Can't dereference NULL

Input Validation

#include int main() { int age; printf("Enter age: "); if (scanf("%d", &age) != 1) { // scanf returns 1 if it read 1 item printf("Invalid input! Must be a number.\n"); return 1; } if (age < 0 || age > 150) { printf("Age out of range!\n"); return 1; } printf("You are %d years old\n", age); return 0; }

6. Organize Code (Multiple Files)

Split your code into .h files (header — declarations) and .c files (source — implementation).

// math_helpers.h — HEADER file #ifndef MATH_HELPERS_H // Include guard — prevents double inclusion #define MATH_HELPERS_H int add(int a, int b); int square(int n); #endif
// math_helpers.c — SOURCE file #include "math_helpers.h" int add(int a, int b) { return a + b; } int square(int n) { return n * n; }
// main.c — the program #include #include "math_helpers.h" // Use "" for YOUR files int main() { printf("%d\n", square(5)); // 25 return 0; } // Compile together: // gcc main.c math_helpers.c -o program
💡 Include guards (#ifndef / #define / #endif) prevent a header from being included twice.
