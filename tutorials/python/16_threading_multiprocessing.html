1. Threading — Temeller

threading.Thread(target=None, name=None, args=(), kwargs={}, daemon=None)
import threading, time def worker(name, delay): print(f" {name}: başladı") time.sleep(delay) print(f" {name}: bitti") t1 = threading.Thread(target=worker, args=("A", 2)) t2 = threading.Thread(target=worker, args=("B", 1)) t1.start(); t2.start() t1.join(); t2.join() print("Ana thread: tümü tamam")
A: başladı B: başladı B: bitti A: bitti Ana thread: tümü tamam

Thread Bilgileri

print(threading.current_thread().name) # "MainThread" print(threading.active_count()) # aktif thread sayısı print(threading.enumerate()) # tüm thread listesi print(t.is_alive()) # çalışıyor mu print(t.ident) # thread ID print(t.native_id) # OS thread ID

2. Thread Sınıfı Alt Sınıflama

class MyThread(threading.Thread): def __init__(self, name, count): super().__init__(name=name) self.count = count def run(self): for i in range(self.count): print(f"{self.name}: {i}"); time.sleep(0.2) ts = [MyThread(f"T-{i}", 3) for i in range(3)] for t in ts: t.start() for t in ts: t.join()

3. Daemon Threads

Ana thread sonlandığında otomatik öldürülür.

def bg(): while True: print(".", end="", flush=True); time.sleep(0.3) t = threading.Thread(target=bg, daemon=True) t.start() time.sleep(1) print("\nAna thread bitiyor")

4. Lock — Senkronizasyon

threading.Lock() # .acquire(blocking, timeout) / .release() / 'with lock:'
counter = 0; lock = threading.Lock() def safe_inc(n): global counter for _ in range(n): with lock: counter += 1 # Non-blocking acquire if lock.acquire(blocking=False): try: print("Lock alındı") finally: lock.release() # Timeout ile if lock.acquire(timeout=3): lock.release()

5. RLock — Reentrant Lock

Aynı thread tarafından birden çok kez alınabilir. Bir fonksiyon başka bir kilitli fonksiyonu çağırdığında deadlock'u önler.

rlock = threading.RLock() def inner(): with rlock: print("Inner") def outer(): with rlock: print("Outer") inner() # ← Lock ile deadlock olurdu! outer()

6. Semaphore

Bir kaynağı aynı anda kullanabilecek maksimum thread sayısını sınırlar.

threading.Semaphore(value=1) # value=3 → 3 thread aynı anda
sem = threading.Semaphore(3) def access_db(n): with sem: print(f" DB-{n}: bağlandı") time.sleep(1) print(f" DB-{n}: çıktı") threads = [threading.Thread(target=access_db, args=(i,)) for i in range(6)] for t in threads: t.start() for t in threads: t.join()

7. Event — Thread Haberleşmesi

Bir thread'in diğer thread'lere sinyal göndermesini sağlar.

threading.Event() # .set() / .clear() / .wait(timeout) / .is_set()
event = threading.Event() def waiter(name): print(f"{name}: bekliyor...") event.wait() print(f"{name}: sinyal alındı!") def signaller(): time.sleep(1) print("Sinyal: SET!") event.set() threading.Thread(target=waiter, args=("W1",)).start() threading.Thread(target=waiter, args=("W2",)).start() threading.Thread(target=signaller).start() # .wait(timeout) — zaman aşımı ile bekle event.wait(timeout=5) # .clear() — sinyali sıfırla event.clear()

8. Condition

Gelişmiş wait/notify pattern. Bir thread koşul gerçekleşene kadar bekler, diğer thread koşulu gerçekleştirip haber verir.

threading.Condition(lock=None) # .wait(timeout) / .notify(n) / .notify_all()
cv = threading.Condition() items = [] def producer(): with cv: for i in range(5): items.append(i) cv.notify(1) # 1 tüketiciyi uyandır time.sleep(0.3) def consumer(): with cv: while True: while not items: cv.wait() # notify gelene kadar bekle item = items.pop(0) print("Tüketildi:", item) if item == 4: break threading.Thread(target=producer).start() threading.Thread(target=consumer).start()

9. Barrier

N thread'in tümü bariyere ulaşana kadar bloke olur, sonra hepsi serbest kalır.

threading.Barrier(n, action=None, timeout=None)
barrier = threading.Barrier(3) def race_car(name): time.sleep(hash(name) % 3) print(f"{name}: bariyerde bekliyor") barrier.wait() print(f"{name}: GO!") cars = [threading.Thread(target=race_car, args=(f"Car {i}",)) for i in range(3)] for c in cars: c.start() for c in cars: c.join() # .abort() — tüm bekleyenlere BrokenBarrierError fırlat # .reset() — bariyeri sıfırla

10. queue.Queue — Thread-Safe Kuyruk

queue.Queue(maxsize=0) # FIFO kuyruk
queue.LifoQueue(maxsize=0) # LIFO (stack)
queue.PriorityQueue(maxsize=0) # öncelik kuyruğu
import queue q = queue.Queue(maxsize=5) # Üretici for i in range(8): q.put(i, block=True, timeout=2) # doluysa bekle # Tüketici item = q.get(block=True, timeout=2) # boşsa bekle q.task_done() # işleme bitti sinyali q.join() # tüm task_done gelene kadar bekle # Non-blocking item = q.get_nowait() # boşsa queue.Empty hatası q.put_nowait(item) # doluysa queue.Full hatası # Kuyruk durumu print(q.qsize()) # eleman sayısı print(q.empty()) # boş mu print(q.full()) # dolu mu

Producer-Consumer Pattern (Tam)

q = queue.Queue(maxsize=5) def producer(): for i in range(8): q.put(i); print(f"Üretildi: {i}"); time.sleep(0.2) def consumer(name): while True: item = q.get() if item is None: break # sinyal: kapat print(f" {name} tüketti: {item}") q.task_done(); time.sleep(0.5) threading.Thread(target=producer).start() threading.Thread(target=consumer, args=("C1",)).start() threading.Thread(target=consumer, args=("C2",)).start() q.join() # tüm öğeler işlenene kadar bekle

11. ThreadPoolExecutor

concurrent.futures modülü ile thread havuzu yönetimi — manuel thread yönetiminden çok daha kolay.

ThreadPoolExecutor(max_workers=None, thread_name_prefix='')

submit() + as_completed()

from concurrent.futures import ThreadPoolExecutor, as_completed def fetch(url): time.sleep(1) return f"Veri: {url}" urls = ["a.com", "b.com", "c.com", "d.com", "e.com"] with ThreadPoolExecutor(max_workers=3) as ex: futures = {ex.submit(fetch, url): url for url in urls} for future in as_completed(futures): print(future.result())

map() — Sıralı sonuç

with ThreadPoolExecutor(4) as ex: results = list(ex.map(fetch, urls)) print(results)

Future Metodları

future = ex.submit(fetch, "test.com") future.result(timeout=5) # sonucu al (bekle) future.done() # tamamlandı mı future.cancel() # iptal et (başlamadıysa) future.cancelled() # iptal edildi mi future.running() # çalışıyor mu future.exception() # varsa exception'ı döndür print(future.add_done_callback(lambda f: print("Bitti!")))

12. Thread-Local Data

Her thread'in kendi kopyasına sahip olduğu veri.

local_data = threading.local() def worker(value): local_data.val = value # her thread kendi kopyasını alır time.sleep(0.1) print(f"{threading.current_thread().name}: {local_data.val}") threads = [threading.Thread(target=worker, args=(i,)) for i in range(3)] for t in threads: t.start() for t in threads: t.join()

14. Multiprocessing — Temeller

14.1 Process Oluşturma

multiprocessing.Process(target=None, name=None, args=(), kwargs={}, daemon=None)
import multiprocessing def cpu_work(n): return sum(i*i for i in range(n)) p = multiprocessing.Process(target=cpu_work, args=(10_000_000,)) p.start() p.join() print("Process bitti")

14.2 Multiprocessing Pool — Pool()

multiprocessing.Pool(processes=None, initializer=None, initargs=())
def square(x): return x * x with multiprocessing.Pool(processes=4) as pool: # map — bloklu results = pool.map(square, range(20)) print(results) # imap — lazy (tek tek iterator döndürür) for r in pool.imap(square, range(20)): print(r, end=" ") # starmap — çoklu argüman r2 = pool.starmap(lambda a, b: a+b, [(1,2), (3,4)])

14.3 Pool.apply() & Pool.apply_async()

def add(a, b): time.sleep(0.5) return a + b with multiprocessing.Pool(4) as pool: # apply — bloklu r1 = pool.apply(add, (10, 20)) # apply_async — non-bloklu result = pool.apply_async(add, (30, 40)) print("Başka iş yapılıyor...") r2 = result.get(timeout=5) # sonucu al # Callback ile async pool.apply_async(add, (5, 6), callback=lambda r: print(f"Callback: {r}")) pool.close() # yeni iş kabul etme pool.join() # tüm işler bitene kadar bekle

16. ProcessPoolExecutor

17. Queue & Pipe (IPC)

17.1 multiprocessing.Queue

q = multiprocessing.Queue() def sender(q): for i in range(5): q.put(i) q.put(None) # bitiş sinyali def receiver(q): while True: item = q.get() if item is None: break print(f"Aldı: {item}") p1 = multiprocessing.Process(target=sender, args=(q,)) p2 = multiprocessing.Process(target=receiver, args=(q,)) p1.start(); p2.start() p1.join(); p2.join()

17.2 Pipe — Çift yönlü haberleşme

multiprocessing.Pipe(duplex=True)
parent, child = multiprocessing.Pipe(duplex=True) def child_proc(conn): conn.send(["Merhaba", "çocuktan"]) msg = conn.recv() print("Çocuk aldı:", msg) conn.close() p = multiprocessing.Process(target=child_proc, args=(child,)) p.start() msg = parent.recv() print("Ebeveyn aldı:", msg) parent.send("Merhaba ebeveynden!") p.join() # conn.poll() — veri var mı kontrol et # conn.send_bytes() — byte verisi gönder # conn.recv_bytes() — byte verisi al

Process'ler arasında paylaşılan bellek (daha hızlı, kopya yok).

18.1 Value

multiprocessing.Value(typecode_or_type, *args, lock=True)
counter = multiprocessing.Value('i', 0) # 'i' = signed int lock = multiprocessing.Lock() def inc(n): for _ in range(n): with lock: counter.value += 1 procs = [multiprocessing.Process(target=inc, args=(10000,)) for _ in range(4)] for p in procs: p.start() for p in procs: p.join() print(counter.value) # 40000

19. Manager — Proxy Nesneler

Manager, farklı process'lerin paylaşılan Python nesnelerine (list, dict, Queue, Lock, vs.) proxy aracılığıyla erişmesini sağlar.

def worker(d, l): d["key"] = "value" l.append(42) with multiprocessing.Manager() as manager: d = manager.dict() # paylaşımlı dict l = manager.list() # paylaşımlı list p = multiprocessing.Process(target=worker, args=(d, l)) p.start(); p.join() print(d) # {'key': 'value'} print(l) # [42] # Manager otomatik kapanır

23. Best Practices

✅ Do:
  • ThreadPoolExecutor / ProcessPoolExecutor kullan — manuel yönetimden kaçın
  • with lock: kullan — manuel acquire/release'den kaçın
  • Queue kullan — thread-safe veri alışverişi için
  • Daemon thread'leri arka plan görevleri için kullan
  • Process havuzu boyutunu CPU çekirdek sayısına eşit yap (os.cpu_count())
  • Kaynakları temizlemek için finally blokları kullan
  • Process'ler arası büyük veri için Manager yerine Queue/Pipe tercih et (daha hızlı)
  • Timeout kullan — asla sonsuz bekleme yapma
❌ Don't:
  • Thread'leri CPU-bound işler için kullanma (GIL darboğazı)
  • Paylaşılan veriyi Lock'suz değiştirme
  • join()'i unutma — main thread erken çıkabilir
  • Çok fazla thread/process oluşturma — havuz kullan
  • multiprocessing'de global değişkenleri değiştirme (her proses kopyasını alır)
  • Pool.map()'e lambda verme — pickle edilemez
  • Lock alırken sırayı bozma — deadlock riski
