Lesson 9: Dictionaries & Sets

Simple English — Based on W3Schools Python Tutorial

Dictionaries — Key-Value Pairs

A dictionary stores data as key: value pairs. Keys must be unique and immutable.

student = { "name": "Alice", "age": 22, "grade": "A" } print(student)
{'name': 'Alice', 'age': 22, 'grade': 'A'}

Accessing Dictionary Items

student = {"name": "Alice", "age": 22} print(student["name"]) # bracket notation print(student.get("age")) # get() method print(student.get("city", "Not Found")) # default value # student["city"] # ❌ KeyError: 'city'
Alice 22 Not Found

Modifying Dictionaries

d = {"a": 1, "b": 2} d["c"] = 3 # add new key-value d["a"] = 99 # update existing key print(d)
{'a': 99, 'b': 2, 'c': 3}

Dictionary Methods

keys() — Get all keys

d = {"x": 1, "y": 2, "z": 3} print(d.keys())
dict_keys(['x', 'y', 'z'])

values() — Get all values

d = {"x": 1, "y": 2, "z": 3} print(d.values())
dict_values([1, 2, 3])

items() — Get all key-value pairs

d = {"name": "Alice", "age": 22} print(d.items()) for key, value in d.items(): print(f"{key} = {value}")
dict_items([('name', 'Alice'), ('age', 22)]) name = Alice age = 22

update() — Merge dictionaries

d1 = {"a": 1, "b": 2} d2 = {"b": 99, "c": 3} d1.update(d2) print(d1)
{'a': 1, 'b': 99, 'c': 3}

pop() — Remove by key (returns the value)

d = {"a": 1, "b": 2, "c": 3} val = d.pop("b") print(f"Removed value: {val}") print(d)
Removed value: 2 {'a': 1, 'c': 3}

popitem() — Remove last inserted item (Python 3.7+)

d = {"a": 1, "b": 2, "c": 3} last = d.popitem() print(f"Removed: {last}") print(d)
Removed: ('c', 3) {'a': 1, 'b': 2}

clear() — Empty the dictionary

d = {"a": 1, "b": 2} d.clear() print(d)
{}

copy() — Shallow copy

original = {"name": "Alice", "age": 22} copied = original.copy() copied["age"] = 30 print("original:", original) print("copied:", copied)
original: {'name': 'Alice', 'age': 22} copied: {'name': 'Alice', 'age': 30}

fromkeys() — Create dict with same value

keys = ["a", "b", "c"] d = dict.fromkeys(keys, 0) print(d)
{'a': 0, 'b': 0, 'c': 0}

setdefault() — Get or set default

d = {"a": 1, "b": 2} val = d.setdefault("c", 99) # key "c" doesn't exist → set to 99 print(val) print(d) val2 = d.setdefault("a", 999) # key "a" exists → returns 1 print(val2)
99 {'a': 1, 'b': 2, 'c': 99} 1

Dictionary Comprehension

squares = {x: x**2 for x in range(5)} print(squares) even_squares = {x: x**2 for x in range(10) if x % 2 == 0} print(even_squares)
{0: 0, 1: 1, 2: 4, 3: 9, 4: 16} {0: 0, 2: 4, 4: 16, 6: 36, 8: 64}

Checking if a Key Exists

d = {"name": "Alice", "age": 22} print("name" in d) print("grade" not in d)
True True

Sets — Unordered, Unique Items

A set is an unordered collection of unique items. You create sets with curly braces {}.

fruits = {"apple", "banana", "cherry", "apple"} # duplicates removed print(fruits)
{'apple', 'banana', 'cherry'}
⚠ Empty {} creates a dictionary, not a set! Use set() for an empty set.

Adding & Removing Set Items

s = {1, 2, 3} s.add(4) # add element print(s) s.remove(2) # raises KeyError if not found print(s) s.discard(5) # no error if not found x = s.pop() # removes and returns an arbitrary element print(f"popped: {x}") print(s) s.clear() # empty the set print(s)
{1, 2, 3, 4} {1, 3, 4} popped: 1 {3, 4} set()

Set Operations

a = {1, 2, 3, 4} b = {3, 4, 5, 6} print("Union:", a | b) # all elements print("Intersection:", a & b) # common elements print("Difference (a - b):", a - b) # in a but not in b print("Symmetric diff:", a ^ b) # in either, but not both
Union: {1, 2, 3, 4, 5, 6} Intersection: {3, 4} Difference (a - b): {1, 2} Symmetric diff: {1, 2, 5, 6}

Set Methods (Alternative Syntax)

a = {1, 2, 3} b = {2, 3, 4} print(a.union(b)) print(a.intersection(b)) print(a.difference(b))
{1, 2, 3, 4} {2, 3} {1}

Subset & Superset

a = {1, 2} b = {1, 2, 3, 4} print(a.issubset(b)) # all elements of a are in b print(b.issuperset(a)) # b contains all elements of a
True True

FrozenSet — Immutable Set

A frozenset is like a set, but immutable. It can be used as a dictionary key.

fs = frozenset([1, 2, 3]) print(fs) # fs.add(4) # ❌ AttributeError: 'frozenset' object has no attribute 'add' # But it works with set operations other = frozenset([2, 3, 4]) print(fs | other) print(fs & other)
frozenset({1, 2, 3}) frozenset({1, 2, 3, 4}) frozenset({2, 3})
