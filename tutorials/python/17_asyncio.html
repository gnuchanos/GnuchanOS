3. await ve asyncio.sleep()

async def fetch_data(name, delay): print(f"{name}: başladı") await asyncio.sleep(delay) # non-blocking bekleme print(f"{name}: bitti") return f"veri-{name}" # asyncio.sleep() vs time.sleep(): # - asyncio.sleep(): event loop diğer görevlere geçer ✅ # - time.sleep(): tüm thread'i bloke eder ❌

4. gather() — Toplu Çalıştırma

asyncio.gather(*coros, return_exceptions=False) → list
async def main(): # 3 task'ı aynı anda çalıştır results = await asyncio.gather( fetch_data("A", 3), fetch_data("B", 1), fetch_data("C", 2), ) print(results) # ['veri-A', 'veri-B', 'veri-C'] # return_exceptions=True — hata fırlatmak yerine exception objesi döndür results = await asyncio.gather(task1, task2, return_exceptions=True)

5. create_task() — Arka Plan Görevleri

asyncio.create_task(coro, *, name=None) → Task
async def main(): # Arka plan görevi oluştur task1 = asyncio.create_task(fetch_data("BG-1", 3), name="background1") task2 = asyncio.create_task(fetch_data("BG-2", 2)) # Başka iş yap print("Ana iş yapılıyor...") await asyncio.sleep(1) print("Ana iş bitti") # Task sonuçlarını bekle r1 = await task1 r2 = await task2 print(r1, r2) # Task bilgileri print("Task adı:", task1.get_name()) print("Bitti mi:", task1.done()) print("İptal edildi mi:", task1.cancelled()) # Task'a yeni isim ver task1.set_name("yeniad") # Not: create_task() bir event loop içinde çağrılmalıdır

6. as_completed() — Sırasız Sonuç

asyncio.as_completed(coros, *, timeout=None)
async def main(): tasks = [fetch_data(f"Task {i}", delay) for i, delay in enumerate([3, 1, 2])] # Sonuçlar bittiği sırayla gelir (sıra garantisi yok) for coro in asyncio.as_completed(tasks): result = await coro print("Tamamlanan:", result)

7. TaskGroup — Yapısal Eşzamanlılık (Python 3.11+)

asyncio.TaskGroup() # async with ile kullanılır
async def main(): async with asyncio.TaskGroup() as tg: t1 = tg.create_task(fetch_data("A", 2)) t2 = tg.create_task(fetch_data("B", 1)) t3 = tg.create_task(fetch_data("C", 3)) # TaskGroup çıkışında tüm task'lar tamamlanmış olur print(f"A: {t1.result()}, B: {t2.result()}, C: {t3.result()}") # Herhangi biri hata fırlatırsa, diğerleri otomatik iptal edilir

8. wait_for() — Zaman Aşımı

asyncio.wait_for(coro, timeout) → result
async def slow(): await asyncio.sleep(10) return "geç" async def main(): try: result = await asyncio.wait_for(slow(), timeout=2) except asyncio.TimeoutError: print("Zaman aşımı!")

9. wait() — Koşullu Bekleme

asyncio.wait(coros, *, timeout=None, return_when=FIRST_COMPLETED) → (done, pending)
async def main(): tasks = {fetch_data("A", 5), fetch_data("B", 2), fetch_data("C", 3)} # İlk tamamlananı bekle done, pending = await asyncio.wait(tasks, return_when=asyncio.FIRST_COMPLETED) print("Tamamlananlar:", len(done)) print("Bekleyenler:", len(pending)) # Bekleyenleri iptal et for t in pending: t.cancel() # return_when seçenekleri: # asyncio.FIRST_COMPLETED — ilk tamamlanan # asyncio.FIRST_EXCEPTION — ilk hata # asyncio.ALL_COMPLETED — tümü (default)

10. Async Context Manager

class AsyncDB: async def __aenter__(self): print("Bağlanılıyor...") await asyncio.sleep(0.5) return self async def __aexit__(self, exc_type, exc_val, exc_tb): print("Bağlantı kapatılıyor...") await asyncio.sleep(0.3) async def query(self, sql): await asyncio.sleep(0.1) return f"sonuç: {sql}" async def main(): async with AsyncDB() as db: result = await db.query("SELECT 1") print(result)

Async Iterator

class AsyncRange: def __init__(self, n): self.n = n; self.i = 0 def __aiter__(self): return self async def __anext__(self): if self.i >= self.n: raise StopAsyncIteration await asyncio.sleep(0.3) self.i += 1 return self.i

12. asyncio.Queue

asyncio.Queue(maxsize=0) # .put() .get() .join() .task_done() .qsize() .empty() .full()

22. Retry & Rate Limiter Pattern

Retry (Exponential Backoff)

async def retry(coro_factory, max_retries=3, delay=1): for attempt in range(max_retries): try: return await coro_factory() except Exception as e: if attempt == max_retries - 1: raise wait = delay * (2 ** attempt) # 1, 2, 4 saniye... print(f"Deneme {attempt+1} başarısız, {wait}s bekleniyor...") await asyncio.sleep(wait)

Rate Limiter (Token Bucket)

class RateLimiter: def __init__(self, rate): self.rate = rate # saniyede izin verilen işlem self.tokens = 0 self.last = time.time() async def acquire(self): now = time.time() self.tokens = min(self.rate, self.tokens + (now - self.last) * self.rate) self.last = now if self.tokens < 1: wait = (1 - self.tokens) / self.rate await asyncio.sleep(wait) self.tokens = 0 else: self.tokens -= 1 # Kullanım: # limiter = RateLimiter(rate=5) # saniyede 5 istek # async with limiter: # await api_call()

25. Best Practices

✅ Do:
  • asyncio.run() kullan — asla manuel event loop yönetimi yapma
  • asyncio.gather() veya TaskGroup kullan — concurrent işlemler için
  • Blocking kütüphaneler (requests, time.sleep) kullanma — aiohttp, asyncio.sleep kullan
  • Zaman aşımı ekle — asyncio.wait_for() veya asyncio.timeout()
  • Semaphore kullan — aşırı yüklenmeyi önlemek için
  • Hataları yakala — task içinde exception'ları ele al
  • Async context manager kullan — kaynak yönetimi için
❌ Don't:
  • Blocking time.sleep() kullanma — asyncio.sleep() kullan
  • Coroutine'i await etmeden çağırma — "coroutine was never awaited" hatası
  • asyncio.run()'ı başka bir event loop içinde çağırma
  • Uzun süren CPU işlerini event loop'ta çalıştırma — ProcessPoolExecutor'a yönlendir
  • Çok fazla task'ı aynı anda oluşturma — Semaphore kullan
  • Task iptalini yakalamamazlık etme — CancelledError her zaman yakalanmalı
  • Senkron lock (threading.Lock) kullanma — asyncio.Lock kullan
