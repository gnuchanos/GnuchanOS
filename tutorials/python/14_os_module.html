Lesson 14: os Module — Operating System

Simple English — Python Built-in Library

The os module gives you access to operating system functions: files, folders, paths, environment variables, system info, and more.

import os

1. Current Directory

os.getcwd() — Get current working directory

print(os.getcwd())
d:\GnuchanOS\tutorials\python

os.chdir(path) — Change current directory

os.chdir("..") print(os.getcwd()) os.chdir("python")
d:\GnuchanOS\tutorials
💡 os.chdir("..") moves up one directory. Always change back if you need the original location.

2. Listing Directory Contents

os.listdir(path='.')

items = os.listdir('.') print(items[:5]) # First 5 entries
['01_intro_getstarted.html', '02_syntax_output.html', '03_variables_datatypes.html', '04_operators.html', '05_conditions.html']

with os.scandir() — Modern, faster

with os.scandir('.') as entries: for entry in entries: if entry.is_file(): print(entry.name, entry.stat().st_size)
01_intro_getstarted.html 2874 02_syntax_output.html 3127 03_variables_datatypes.html 3266 ...
💡 os.scandir() returns DirEntry objects with methods like .is_file(), .is_dir(), .stat() — much faster than os.listdir() for large directories.

3. Folder Operations

os.mkdir(path) — Single folder

try: os.mkdir("temp_folder") print("Created temp_folder") except FileExistsError: print("Already exists")
Created temp_folder

os.makedirs(path) — Nested folders (like mkdir -p)

os.makedirs("a/b/c/d", exist_ok=True) print("Created a/b/c/d")
Created a/b/c/d

os.rmdir(path) — Remove empty folder

os.rmdir("temp_folder") print("Deleted temp_folder")
Deleted temp_folder

shutil.rmtree(path) — Remove folder with all contents (⚠️ DANGER!)

import shutil shutil.rmtree("a") print("Deleted a and all its subfolders")
Deleted a and all its subfolders
⚠️ shutil.rmtree() permanently deletes the entire directory tree — there is no recycle bin. Use with extreme caution!

4. File Operations

os.rename(src, dst) — Rename or move a file

# Create a dummy file first, then rename it open("old_name.txt", "w").close() os.rename("old_name.txt", "new_name.txt") print("Renamed to new_name.txt")
Renamed to new_name.txt

os.replace(src, dst) — Rename/move, overwrites destination

open("target.txt", "w").close() os.replace("new_name.txt", "target.txt") print("Replaced target.txt")
Replaced target.txt

os.remove(path) — Delete a file (PERMANENT!)

os.remove("target.txt") print("Deleted target.txt")
Deleted target.txt

os.removedirs(path) — Remove empty folder chain

os.makedirs("x/y/z", exist_ok=True) os.removedirs("x/y/z") print("Removed x/y/z chain")
Removed x/y/z chain
⚠️ os.remove() deletes files permanently — they do not go to the recycle bin. os.removedirs() only works if all directories are empty.

5. File Information

os.path.exists(path)

print(os.path.exists("13_advanced_topics.html")) print(os.path.exists("does_not_exist.txt"))
True False

os.path.isfile / isdir

print(os.path.isfile("13_advanced_topics.html")) print(os.path.isdir("13_advanced_topics.html")) print(os.path.isdir("."))
True False True

os.path.getsize(path) — File size in bytes

size = os.path.getsize("13_advanced_topics.html") print(f"Size: {size} bytes")
Size: 7268 bytes

os.path.getmtime / getctime — Timestamps

import time mtime = os.path.getmtime("13_advanced_topics.html") ctime = os.path.getctime("13_advanced_topics.html") print("Modified:", time.ctime(mtime)) print("Created:", time.ctime(ctime))
Modified: Mon Feb 10 10:00:00 2025 Created: Mon Feb 10 10:00:00 2025

6. Path Operations

os.path.join(a, b) — Smart path joining (OS-aware)

path = os.path.join("folder", "sub", "file.txt") print(path)
folder\sub\file.txt
💡 On Windows it uses backslashes, on Linux/macOS forward slashes — automatically!

os.path.dirname / basename / splitext

p = r"d:\projects\data\report.csv" print("Dir:", os.path.dirname(p)) print("File:", os.path.basename(p)) print("Split ext:", os.path.splitext(p))
Dir: d:\projects\data File: report.csv Split ext: ('d:\\projects\\data\\report', '.csv')

os.path.abspath / realpath

print(os.path.abspath("14_os_module.html")) print(os.path.realpath("14_os_module.html"))
d:\GnuchanOS\tutorials\python\14_os_module.html d:\GnuchanOS\tutorials\python\14_os_module.html

os.path.split / expanduser

path = r"d:\data\docs\report.pdf" print(os.path.split(path)) print(os.path.expanduser("~"))
('d:\\data\\docs', 'report.pdf') C:\Users\btw

7. Environment Variables

os.environ — Dictionary of all env vars

print(os.environ.get("PATH")[:60]) # First 60 chars
C:\Windows\system32;C:\Windows;C:\Program Files\Python312\Scripts\...

os.environ.get / os.getenv

home = os.environ.get("HOME", "Not set") user = os.getenv("USERNAME", "Unknown") print("Home:", home) print("User:", user)
Home: Not set User: btw

Setting environment variables

os.environ["MY_VAR"] = "hello" print(os.getenv("MY_VAR"))
hello
⚠️ Environment variables set via os.environ only affect the current process — they are not permanent system-wide changes.

8. System Information

os.name — OS name ('nt' or 'posix')

print(os.name)
nt

os.cpu_count() — Number of CPU cores

print("CPU cores:", os.cpu_count())
CPU cores: 8

os.uname() — Detailed system info (Unix only)

# On Linux/macOS: # print(os.uname()) print("Only available on Unix systems")
Only available on Unix systems

os.getlogin() — Current user name

print("Logged in as:", os.getlogin())
Logged in as: btw

9. Directory Walking

os.walk(top) — Recursively traverse directories

for dirpath, dirnames, filenames in os.walk("."): print("Dir:", dirpath) print(" Files:", filenames[:3]) break # Just show the first level
Dir: . Files: ['01_intro_getstarted.html', '02_syntax_output.html', '03_variables_datatypes.html', ...]
💡 os.walk() is perfect for recursively processing all files in a directory tree. Modifying dirnames in-place lets you prune the walk.

10. Running Commands (Legacy)

os.system('cmd') — Run command, no output capture

exit_code = os.system("echo Hello from cmd") print("Exit code:", exit_code)
Hello from cmd Exit code: 0

os.popen('cmd').read() — Capture output

result = os.popen("echo Hello from popen").read() print(result)
Hello from popen
💡 For modern Python, use subprocess.run() instead of os.system() or os.popen().

11. shutil — Higher-Level File Operations

import shutil

shutil.copy(src, dst) — Copy a file

# Create a source file first open("source.txt", "w").write("Hello\n") shutil.copy("source.txt", "dest.txt") print("Copied") os.remove("source.txt") os.remove("dest.txt")
Copied

shutil.copytree(src, dst) — Copy entire directory tree

os.makedirs("backup/original", exist_ok=True) open("backup/original/readme.txt", "w").close() shutil.copytree("backup", "backup_copy") print("Copied backup to backup_copy") shutil.rmtree("backup") shutil.rmtree("backup_copy")
Copied backup to backup_copy

shutil.move(src, dst) — Move/rename

open("move_me.txt", "w").close() shutil.move("move_me.txt", "moved.txt") print("Moved") os.remove("moved.txt")
Moved

shutil.rmtree(path) — Delete directory tree (already covered)

shutil.disk_usage(path) — Disk usage statistics

usage = shutil.disk_usage(".") print(f"Total: {usage.total // (1024**3)} GB") print(f"Used: {usage.used // (1024**3)} GB") print(f"Free: {usage.free // (1024**3)} GB")
Total: 256 GB Used: 182 GB Free: 74 GB
