type() — Type Checking

print(type(nil)) -- nil
print(type(true)) -- boolean
print(type(42)) -- number
print(type("hello")) -- string
print(type({})) -- table
print(type(print)) -- function

Operators

Aritmetik

print(2 ^ 10) -- 1024.0 (üs)
print(15 // 4) -- 3 (floor division)
print(10 % 3) -- 1 (mod)

Karşılaştırma

print(10 == 10) -- true
print(10 ~= 20) -- true (not equal)
print("a" < "b") -- true (lexicographic)

Mantıksal

print(true and false) -- false
print(true or false) -- true
local name = nil
print(name or "Guest")

String Concatenation

print("Hello" .. " " .. "World")

Truthy & Falsy

Sadece false ve nil falsy'dir. 0 ve boş string TRUE!

if 0 then print("0 is true") end -- prints!
if "" then print("empty str is true") end
