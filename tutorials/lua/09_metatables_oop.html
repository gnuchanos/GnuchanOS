Metatables

Metatable, bir tablonun davranisini degistirir. Metamethodlarla operator overloading.

local Vec = {
__add = function(a, b)
return {x = a.x + b.x, y = a.y + b.y}
end,
__tostring = function(v)
return string.format("(%d,%d)", v.x, v.y)
end
}
local v1 = setmetatable({x=1,y=2}, Vec)
local v2 = setmetatable({x=3,y=4}, Vec)
print(v1 + v2)

__index Inheritance

local Animal = {speak = function(self)
print("My name is " .. self.name)
end}
local Dog = {name = "Rex"}
setmetatable(Dog, {__index = Animal})
Dog:speak()

OOP Pattern

local Person = {__index = Person}
function Person:new(name, age)
local s = setmetatable({}, Person)
s.name = name; s.age = age
return s
end
function Person:greet()
print("Hi, I'm " .. self.name)
end
local p = Person:new("Ali", 30)
p:greet()
