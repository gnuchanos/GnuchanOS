Coroutine Basics

Coroutine'ler kooperatif multitasking saglar. Fonksiyon calisirken yield ile bekler, sonra kaldigi yerden devam eder.

local co = coroutine.create(function()
for i = 1, 3 do
print("coro: " .. i)
coroutine.yield(i * 10)
end
return "done"
end)

-- coroutine.resume() returns: status, yield values
print(coroutine.resume(co)) -- true 10
print(coroutine.resume(co)) -- true 20
print(coroutine.resume(co)) -- true 30
print(coroutine.resume(co)) -- false dead

Coroutine Status

local co = coroutine.create(function()
coroutine.yield()
end)

print(coroutine.status(co)) -- suspended
coroutine.resume(co)
print(coroutine.status(co)) -- dead
