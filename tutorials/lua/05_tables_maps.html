Lua Tutorial — GnuchanOS

Key-Value Tables

local person = {
name = "Ali",
age = 25,
city = "Istanbul"
}

print(person.name) -- Ali
print(person["name"]) -- Ali
person.job = "Engineer"

for k, v in pairs(person) do
print(k, v)
end

Nested Tables

local school = {
name = "ABC School",
students = {
{name = "Ali", grade = 85},
{name = "Ayse", grade = 92}
}
}
print(school.students[1].name) -- Ali
