💜 Lesson 6: Control Flow

Lua Tutorial — GnuchanOS

if/elseif/else

local age = 20
if age < 18 then
print("Minor")
elseif age < 65 then
print("Adult")
else
print("Senior")
end

for Loop

-- Numeric for: start, stop, step
for i = 1, 5 do print(i) end

-- Geri sayim
for i = 10, 1, -2 do print(i) end

-- Generic for (table)
for i, v in ipairs({"a","b","c"}) do
print(i, v)
end

while & repeat

-- while
local i = 1
while i <= 5 do
print(i)
i = i + 1
end

-- repeat (en az 1 kez calisir)
local x = 10
repeat
print(x)
x = x - 1
until x == 0

break & goto

-- break
for i = 1, 10 do
if i > 5 then break end
print(i)
end

-- goto (continue benzeri)
for i = 1, 10 do
if i % 2 == 0 then goto continue end
print(i) -- 1 3 5 7 9
::continue::
end
